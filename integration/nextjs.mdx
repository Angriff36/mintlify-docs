This guide walks you through adding Manifest to an existing Next.js App Router project. By the end, you will have a `.manifest` domain model, generated route handlers for reads and writes, and Clerk auth wired into the identity pipeline.

## Prerequisites

- Next.js 13+ with the App Router
- Node.js 18+
- An existing Next.js project (or `npx create-next-app@latest`)

## Workflow

<Steps>
  <Step title="Install the package">
    Add `@angriff36/manifest` to your project:

    ```bash
    npm install @angriff36/manifest
    ```
  </Step>

  <Step title="Write your .manifest file">
    Create a file describing your domain model. This example models a `Recipe` entity with a `publish` command:

    ```manifest
    entity Recipe {
      property required id: string
      property required name: string
      property category: string?
      property status: string = "draft"
      property authorId: string

      command publish() {
        guard self.status == "draft"
        guard user.id == self.authorId
        mutate status = "published"
        emit RecipePublished
      }

      policy canPublish execute: user.role in ["author", "editor"]
    }

    store Recipe in memory

    event RecipePublished: "kitchen.recipe.published" {
      recipeId: string
      authorId: string
    }
    ```

    Save this as `recipe.manifest` in your project root.
  </Step>

  <Step title="Generate routes">
    Use the projection CLI to generate the route files. Run these commands from your project root:

    ```bash
    # Compile the .manifest file to IR
    manifest compile recipe.manifest --output ir/

    # Generate all surfaces from the compiled IR
    manifest generate ir/recipe.ir.json \
      --auth clerk \
      --output app/api/

    # Or generate specific surfaces individually
    manifest generate ir/recipe.ir.json --surface types --output lib/manifest-types.ts
    manifest generate ir/recipe.ir.json --surface client --output lib/manifest-client.ts
    ```

    <Note>
      The CLI is a development scaffolding tool. For automated build pipelines, call the projection API programmatically in a build script instead.
    </Note>
  </Step>

  <Step title="Review the generated read route">
    The generated GET handler uses a direct database query — it does not call `RuntimeEngine`. This is intentional: read policies are not enforced at the query level by default, and direct queries are more efficient.

    ```typescript
    // app/api/recipes/route.ts — auto-generated, illustrative
    import { NextRequest } from "next/server";
    import { auth } from "@clerk/nextjs/server";
    import { database } from "@/lib/database";
    import { manifestSuccessResponse, manifestErrorResponse } from "@/lib/manifest-response";

    export async function GET(request: NextRequest) {
      try {
        const { userId } = await auth();
        if (!userId) {
          return manifestErrorResponse("Unauthorized", 401);
        }

        const userMapping = await database.userTenantMapping.findUnique({
          where: { userId },
        });

        if (!userMapping) {
          return manifestErrorResponse("User not mapped to tenant", 400);
        }

        const { tenantId } = userMapping;

        // Direct Prisma query — bypasses runtime deliberately
        const recipes = await database.recipe.findMany({
          where: { tenantId, deletedAt: null },
          orderBy: { createdAt: "desc" },
        });

        return manifestSuccessResponse({ recipes });
      } catch (error) {
        console.error("Error fetching recipes:", error);
        return manifestErrorResponse("Internal server error", 500);
      }
    }
    ```
  </Step>

  <Step title="Review the generated command route">
    The generated POST handler **must** call `RuntimeEngine.runCommand`. Guards, policies, constraints, and event emissions only run through the runtime.

    The generated handler strips client-supplied identity fields (`id`, `userId`, `tenantId`, `orgId`, `user`) from the request body and injects authoritative values from Clerk and the URL path before calling the runtime:

    ```typescript
    // app/api/recipes/commands/publish/route.ts — auto-generated, illustrative
    import { NextRequest } from "next/server";
    import { auth } from "@clerk/nextjs/server";
    import { RuntimeEngine } from "@angriff36/manifest";
    import { ir } from "@/lib/manifest-ir";
    import { manifestSuccessResponse, manifestErrorResponse } from "@/lib/manifest-response";

    export async function POST(
      request: NextRequest,
      { params }: { params: { id: string } }
    ) {
      const { userId } = await auth();
      if (!userId) {
        return manifestErrorResponse("Unauthorized", 401);
      }

      const body = await request.json();

      // Strip client-supplied identity fields — never trust these
      const {
        id: _id,
        userId: _userId,
        tenantId: _tenantId,
        orgId: _orgId,
        user: _user,
        ...safeInput
      } = body;

      const runtime = new RuntimeEngine(ir, {
        user: { id: userId, role: 'author' },
      });

      const result = await runtime.runCommand(
        'publish',
        { ...safeInput },
        { entityName: 'Recipe', instanceId: params.id }
      );

      if (!result.success) {
        return manifestErrorResponse(result.error ?? 'Command failed', 422);
      }

      return manifestSuccessResponse({ events: result.events });
    }
    ```
  </Step>

  <Step title="Wire up Clerk auth">
    If you chose `--auth clerk`, install the Clerk SDK and configure your middleware:

    ```bash
    npm install @clerk/nextjs
    ```

    Add `clerkMiddleware` to `middleware.ts`:

    ```typescript
    import { clerkMiddleware } from '@clerk/nextjs/server';

    export default clerkMiddleware();

    export const config = {
      matcher: ['/((?!_next|[^?]*\\.(?:html?|css|js(?!on)|jpe?g|webp|png|gif|svg|ttf|woff2?|ico|csv|docx?|xlsx?|zip|webmanifest)).*)', '/(api|trpc)(.*)'],
    };
    ```

    Set your environment variables in `.env.local`:

    ```bash
    NEXT_PUBLIC_CLERK_PUBLISHABLE_KEY=pk_...
    CLERK_SECRET_KEY=sk_...
    ```

    The generated routes call `auth()` from `@clerk/nextjs/server` to resolve `userId`. No additional configuration is needed in the route files.
  </Step>
</Steps>

## Multi-tenant setup

To enable per-tenant data isolation, keep a `userTenantMapping` table that maps `userId` to `tenantId`. The generated read routes look up this mapping before querying:

```typescript
const userMapping = await database.userTenantMapping.findUnique({
  where: { userId },
});

if (!userMapping) {
  return manifestErrorResponse("User not mapped to tenant", 400);
}

const { tenantId } = userMapping;

const recipes = await database.recipe.findMany({
  where: { tenantId, deletedAt: null },
});
```

Pass `tenantId` to the projection generator to enable filtering:

```bash
manifest generate ir/recipe.ir.json --auth clerk --output app/api/
# Default: includeTenantFilter=true, tenantIdProperty='tenantId'
```

To use a different property name (e.g., `orgId`):

```typescript
// In a build script using the programmatic API
const result = projection.generateRoute(ir, 'Recipe', {
  authProvider: 'clerk',
  tenantIdProperty: 'orgId',
});
```

## Workspace / monorepo setup

In a monorepo, it is common to compile the IR once and share it across packages. A typical layout:

```
apps/
  web/           # Next.js app — consumes generated routes and client SDK
packages/
  domain/        # recipe.manifest, build script, generated IR artifact
  api-types/     # generated ts.types output, shared across apps
```

In `packages/domain`, add a build script that compiles the IR and runs projections:

```typescript
// packages/domain/scripts/build.ts
import { compileToIR } from '@angriff36/manifest/ir-compiler';
import { NextJsProjection } from '@angriff36/manifest/projections/nextjs';
import fs from 'fs/promises';
import path from 'path';

const source = await fs.readFile('recipe.manifest', 'utf-8');
const { ir } = await compileToIR(source);
if (!ir) process.exit(1);

// Write IR artifact for apps to import
await fs.writeFile('dist/recipe.ir.json', JSON.stringify(ir, null, 2));

const projection = new NextJsProjection();

// Write types to the shared types package
const typesResult = projection.generateTypes(ir);
await fs.writeFile(
  path.resolve('../../packages/api-types/src/recipe.ts'),
  typesResult.code
);

// Write routes into the Next.js app
const routeResult = projection.generateRoute(ir, 'Recipe', { authProvider: 'clerk' });
await fs.writeFile(
  path.resolve('../../apps/web/app/api/recipes/route.ts'),
  routeResult.code
);
```

In `apps/web`, import the compiled IR directly:

```typescript
// apps/web/lib/manifest-ir.ts
import ir from '@repo/domain/dist/recipe.ir.json';
export { ir };
```

## Listening to events in route handlers

Register an `onEvent` listener before calling `runCommand` to tap into emitted events. This is useful for publishing to an outbox or triggering webhooks:

```typescript
const runtime = new RuntimeEngine(ir, {
  user: { id: userId, role: 'author' },
});

const unsubscribe = runtime.onEvent((event) => {
  console.log(event.name, event.channel, event.payload);
  // Enqueue to your outbox, publish to Pub/Sub, etc.
});

try {
  const result = await runtime.runCommand(
    'publish',
    {},
    { entityName: 'Recipe', instanceId: params.id }
  );

  if (!result.success) {
    return manifestErrorResponse(result.error ?? 'Command failed', 422);
  }

  return manifestSuccessResponse({ events: result.events });
} finally {
  unsubscribe();
}
```

## Common issues

<AccordionGroup>
  <Accordion title='"Entity not found in IR"'>
    Entity names are case-sensitive. Verify that the name you pass to `runCommand` exactly matches the entity name declared in your `.manifest` file.
  </Accordion>

  <Accordion title="Guard fails even with the correct user">
    Guards reference `user.*` from the runtime context, not from the request body. Make sure you pass `user: { id: userId, role: '...' }` when constructing `RuntimeEngine`. If the guard references `user.role` and you did not supply it, the failure is correct behavior.
  </Accordion>

  <Accordion title="Generated route imports are wrong">
    The generated code uses import paths like `@/lib/database` and `@/lib/manifest-response`. These are conventions — create those modules in your project or pass custom `databaseImportPath` and `responseImportPath` options to the projection generator.
  </Accordion>

  <Accordion title="Tenant filtering returns no results">
    Confirm that your database rows have a `tenantId` column populated and that the `userTenantMapping` lookup is returning the correct value. You can disable tenant filtering during development by passing `includeTenantFilter: false` to the generator.
  </Accordion>
</AccordionGroup>
